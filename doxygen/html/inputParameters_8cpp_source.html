162 bool
163 inputParameters::configureFromFile(const std::string &_configFileName)
164 {
165 
166  int nParameters = _ip.parseFile(_configFileName);
167 
168  if(nParameters == -1)
169  {
170  printWarn << "could not open file '" << _configFileName << "'" << endl;
171  return false;
172  }
173 
174 
175  if(_ip.validateParameters(cerr))
176  printInfo << "successfully read input parameters from '" << _configFileName << "'" << endl;
177  else {
178  printWarn << "problems reading input parameters from '" << _configFileName << "'" << endl
179  << *this;
180  return false;
181  }
182  return true;
183 }
