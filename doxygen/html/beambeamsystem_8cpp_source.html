111 double
112 beamBeamSystem::probabilityOfBreakup(const double D) const
113 {
114 
115  double bMin = (_beam1.nuclearRadius()+_beam2.nuclearRadius())/2.;
116  double pOfB = 0.; // PofB = 1 means that there will be a UPC event and PofB = 0 means no UPC
117 
118  // Do pp here
119  if ((_beam1.Z() == 1) && (_beam1.A() == 1) && (_beam2.Z() == 1) && (_beam2.A() == 1)) {
120  double ppslope=19.8;
121  double GammaProfile = exp(-D * D / (2. * hbarc * hbarc * ppslope));
122  pOfB = (1. - GammaProfile) * (1. - GammaProfile);
123  return pOfB;
124  }
125  else if( _beam1.A() == 0 || _beam2.A() == 0 ) // Do eX collisions here. Assume user hasn't used two electrons (A()= 0).
126  return 1.;
127  else if ( ( (_beam1.A() == 1) && (_beam2.A() != 1) ) || ((_beam1.A() != 1) && (_beam2.A() == 1)) ) {
128  // This is pA
129  if( _beam1.A() == 1 ){
130  bMin = _beam2.nuclearRadius() + 0.7;
131  pOfB = exp(-7.0*_beam2.rho0()*_beam2.thickness(D));
132  }else if( _beam2.A() == 1 ){
133  bMin = _beam1.nuclearRadius() + 0.7;
134  pOfB = exp(-7.0*_beam1.rho0()*_beam1.thickness(D));
135  }else{
136  cout<<"Some logical problem here!"<<endl;
137  }
138  return pOfB;
139 
140  }
141 
142  //use the lookup table and return...
143  pOfB = 1.;
144  if (D > 0.0) {
145  //Now we must determine which step number in d corresponds to this D,
146  // and use appropiate Ptot(D_i)
147  int i = (int)(log(D / bMin) / log(_breakupImpactParameterStep));
148  if (i <= 0)
149  pOfB = _breakupProbabilities[0];
150  else{
151  if (i >= int(_breakupProbabilities.size()-1))
152  pOfB = _breakupProbabilities[_breakupProbabilities.size()-1];
153  else {
154  const double DLow = bMin * pow((_breakupImpactParameterStep), i);
155  const double DeltaD = (_breakupImpactParameterStep-1) * DLow;
156  const double DeltaP = _breakupProbabilities[i + 1] - _breakupProbabilities[i];
157  pOfB = _breakupProbabilities[i] + DeltaP * (D - DLow) / DeltaD;
158  }
159  }
160  }
161 
162  return pOfB;
163 }

165 void
166 beamBeamSystem::generateBreakupProbabilities()
167 {
168 
169  double bMin = (_beam1.nuclearRadius()+_beam2.nuclearRadius())/2.;
170 
171  // Do this only for nucleus-nucleus collisions.
172  // pp and pA are handled directly in probabilityOfBreakup
173 // if ((_beam1.Z() != 1) && (_beam1.A() != 1) && (_beam2.Z() != 1) && _beam2.A() != 1) {
174 // this change allows deuterium and tritium to be handled here
175  if ((_beam1.Z() != 1) && (_beam1.A() != 1) && _beam2.A() != 1) {
176 
177  if (_beamBreakupMode == 1)
178  printInfo << "Hard Sphere Break criteria. b > " << 2. * _beam1.nuclearRadius() << endl;
179  if (_beamBreakupMode == 2)
180  printInfo << "Requiring XnXn [Coulomb] breakup. " << endl;
181  if (_beamBreakupMode == 3)
182  printInfo << "Requiring 1n1n [Coulomb only] breakup. " << endl;
183  if (_beamBreakupMode == 4)
184  printInfo << "Requiring both nuclei to remain intact. " << endl;
185  if (_beamBreakupMode == 5)
186  printInfo << "Requiring no hadronic interactions. " << endl;
187  if (_beamBreakupMode == 6)
188  printInfo << "Requiring breakup of one or both nuclei. " << endl;
189  if (_beamBreakupMode == 7)
190  printInfo << "Requiring breakup of one nucleus (Xn,0n). " << endl;
191 
192  double pOfB = 0;
193  double b = bMin;
194  double totRad = _beam1.nuclearRadius()+_beam2.nuclearRadius();
195 
196  while(1)
197  {
198 
199  if(_beamBreakupMode != 5)
200  {
201  if(b > (totRad*1.5))
202  {
203  if(pOfB<_breakupCutOff)
204  {
205 // std::cout << "Break off b: " << b << std::endl;
206 // std::cout << "Number of PofB bins: " << _breakupProbabilities.size() << std::endl;
207  break;
208  }
209  }
210  }
211  else
212  {
213  if((1-pOfB)<_breakupCutOff)
214  {
215 // std::cout << "Break off b: " << b << std::endl;
216 // std::cout << "Number of PofB bins: " << _breakupProbabilities.size() << std::endl;
217  break;
218  }
219  }
220 // std::cout << 1-pOfBreakup << std::endl;
221 
222  probabilityOfHadronBreakup(b);
223  probabilityOfPhotonBreakup(b, _beamBreakupMode);
224 
225  //What was probability of photonbreakup depending upon mode selection,
226  // is now done in the photonbreakupfunction
227  if (_beamBreakupMode == 1) {
228  if (b >_beam1.nuclearRadius()+_beam2.nuclearRadius()) // symmetry
229  _pHadronBreakup = 0;
230  else
231  _pHadronBreakup = 999.;
232  }
233 
234  b *= _breakupImpactParameterStep;
235  pOfB = exp(-1 * _pHadronBreakup) * _pPhotonBreakup;
236  _breakupProbabilities.push_back(pOfB);
237  } // End while(1)
238  }
239

243 double
244 beamBeamSystem::probabilityOfHadronBreakup(const double impactparameter)
245 {
246  //probability of hadron breakup,
247  //this is what is returned when the function is called
248  double gamma = _beamLorentzGamma;
249  //input for gamma_em
250  double b = impactparameter;
251  int a1 = _beam1.A();
252  int a2 = _beam2.A();
253 
254  static int IFIRSTH = 0;
255  static double DELL=0., DELR=0., SIGNN=0., R1=0., A1=0., A2=0., R2=0., RHO1=0.;
256  static double RHO2=0., NZ1=0., NZ2=0., NR1=0., NR2=0.,RR1=0., RR2=0., NY=0., NX=0.;
257  static double AN1=0., AN2=0.;
258  double RSQ=0.,Z1=0.,Z2=0.,Y=0.,X=0.,XB=0.,RPU=0.,IRUP=0.,RTU=0.;
259  double IRUT=0.,T1=0.,T2=0.;
260  static double DEN1[20002], DEN2[20002];
261  double energy,sigmainmb;
262  if (IFIRSTH != 0) goto L100;
263  //Initialize
264  //Integration delta x, delta z
265  IFIRSTH = 1;
266  DELL = .05;
267  DELR = .01;
268 
269  // replace this with a parameterization from the particle data book SRK 4/1025
270  //use two sigma_NN's. 52mb at rhic 100gev/beam, 88mb at LHC 2.9tev/beam, gamma is in cm system
271  //SIGNN = 5.2;
272  //if ( gamma > 500. ) SIGNN = 8.8;
273  energy=2*gamma*0.938; // center of mass energy, in GeV
274  // This equation is from section 50 of the particle data book, the subsection on "Total Hadronic Cross-Sections, using the parameterization for sqrt{s} > 7 GeV.
275  // only the first and second terms contribute significantly, but leave them all here for good measure
276  sigmainmb = 0.2838*pow(log(energy),2)+33.73+13.67*pow(energy,-0.412)-7.77*pow(energy,-0.5626);
277  SIGNN=sigmainmb/10.;
278 
279  //use parameter from Constants
280  R1 = ( _beam1.nuclearRadius());
281  R2 = ( _beam2.nuclearRadius());
282  A1 = 0.535; //This is woodsaxonskindepth
283  A2 = 0.535;
284  //write(6,12)r1,a1,signn Here is where we could probably set this up asymmetrically R2=_beam2.nuclearRadius() and RHO2=ap2=_beam2.A()
285  // R2 = R1;
286  RHO1 = a1;
287  RHO2 = a2;
288  NZ1 = ((R1+5.)/DELR);
289  NR1 = NZ1;
290  NZ2 = ((R2+5.)/DELR);
291  NR2 = NZ2;
292  RR1 = -DELR;
293  RR2 = -DELR;
294  NY = ((R1+5.)/DELL);
295  NX = 2*NY;
296  // This calculates T_A(b) for beam 1 and stores it in DEN1[IR1]
297  for ( int IR1 = 1; IR1 <= NR1; IR1++) {
298  DEN1[IR1] = 0.;
299  RR1 = RR1+DELR;
300  Z1 = -DELR/2;
301 
302  for ( int IZ1 = 1; IZ1 <= NZ1; IZ1++) {
303  Z1 = Z1+DELR;
304  RSQ = RR1*RR1+Z1*Z1;
305  DEN1[IR1] = DEN1[IR1]+1./(1.+exp((sqrt(RSQ)-R1)/A1));
306  }
307 
308  DEN1[IR1] = DEN1[IR1]*2.*DELR;
309  }
310 
311  // This calculates T_A(b) for beam 2 and stores it in DEN2[IR2]
312  for ( int IR2 = 1; IR2 <= NR2; IR2++) {
313  DEN2[IR2] = 0.;
314  RR2 = RR2+DELR;
315  Z2 = -DELR/2;
316 
317  for ( int IZ2 = 1; IZ2 <= NZ2; IZ2++) {
318  Z2 = Z2+DELR;
319  RSQ = RR2*RR2+Z2*Z2;
320  DEN2[IR2] = DEN2[IR2]+1./(1.+exp((sqrt(RSQ)-R2)/A2));
321  }
322 
323  DEN2[IR2] = DEN2[IR2]*2.*DELR;
324  }
325 
326  AN1 = 0.;
327  RR1 = 0.;
328  RR2 = 0.;
329 
330  for ( int IR1 =1; IR1 <= NR1; IR1++) {
331  RR1 = RR1+DELR;
332  AN1 = AN1+RR1*DEN1[IR1]*DELR*2.*starlightConstants::pi;
333  }
334  for ( int IR2 =1; IR2 <= NR2; IR2++) {
335  RR2 = RR2+DELR;
336  AN2 = AN2+RR2*DEN2[IR2]*DELR*2.*starlightConstants::pi;
337  }
338 
339 
340  //.1 to turn mb into fm^2
341  //Calculate breakup probability here
342  L100:
343  _pHadronBreakup = 0.;
344  if ( b > 25. ) return _pHadronBreakup;
345  Y = -.5*DELL;
346  for ( int IY = 1; IY <= NY; IY++) {
347  Y = Y+DELL;
348  X = -DELL*float(NY+1);
349 
350  for ( int IX = 1; IX <=NX; IX++) {
351  X = X+DELL;
352  XB = b-X;
353  RPU = sqrt(X*X+Y*Y);
354  IRUP = (RPU/DELR)+1;
355  RTU = sqrt(XB*XB+Y*Y);
356  IRUT = (RTU/DELR)+1;
357  T1 = DEN2[(int)IRUT]*RHO2/AN2;
358  T2 = DEN1[(int)IRUP]*RHO1/AN1;
359  //Eq.6 BCW, Baltz, Chasman, White, Nucl. Inst. & Methods A 417, 1 (1998)
360  _pHadronBreakup=_pHadronBreakup+2.*T1*(1.-exp(-SIGNN*T2))*DELL*DELL;
361  }//for(IX)
362  }//for(IY)
363 
364  return _pHadronBreakup;
365 }
